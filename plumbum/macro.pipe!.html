Provides a convient syntax for conduit operations.

Example

#[macro_use] extern crate plumbum;
use plumbum::*;
fn main() {
    let src = pipe!{
        produce(42);
        produce(43);
    };
    let sink = pipe!{
        for x = consume();
        for y = consume();
        return x.unwrap_or(0) + y.unwrap_or(0)
    };
    assert_eq!(src.connect(sink), 85);
}
